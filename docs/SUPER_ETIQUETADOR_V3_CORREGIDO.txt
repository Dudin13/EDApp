import os
import uuid
import tempfile
import cv2
from flask import Flask, send_from_directory, request, jsonify, after_this_request
from flask_cors import CORS
from pathlib import Path
from ultralytics import YOLO
from ultralytics import SAM
import torch

app = Flask(__name__)

# ✅ CORS restringido a localhost (antes era abierto a cualquier origen)
CORS(app, origins=["http://localhost:*", "http://127.0.0.1:*"])

# ✅ Paths usando variable de entorno para portabilidad
BASE = Path(os.environ.get("APPED_ROOT", "C:/apped"))
DATASET_ROOT = (BASE / "04_Datasets_Entrenamiento/hybrid_dataset").absolute()
MODEL_PATH = BASE / "train_yolo/runs/detect/EDudin_v1/weights/best.pt"
if not MODEL_PATH.exists():
    MODEL_PATH = BASE / "yolo11m-seg.pt"

# ✅ Subsets válidos - whitelist para evitar path traversal
VALID_SUBSETS = {"train", "val", "test"}

print(f"Dataset Root: {DATASET_ROOT}")
model = YOLO(MODEL_PATH)

# ✅ SAM con carga lazy - no ocupa memoria hasta que se usa
_sam_model = None

def get_sam_model():
    global _sam_model
    if _sam_model is None:
        print("🔄 Cargando modelo SAM...")
        _sam_model = SAM("sam2_b.pt")
        print("✅ SAM cargado.")
    return _sam_model


# ✅ Función de validación de paths para evitar path traversal
def safe_resolve(base: Path, *parts) -> Path | None:
    try:
        resolved = (base.joinpath(*parts)).resolve()
        if not str(resolved).startswith(str(base.resolve())):
            return None
        return resolved
    except Exception:
        return None


@app.route('/api/segment', methods=['POST'])
def segment_image():
    data = request.json
    subset = data.get('subset')
    filename = data.get('filename')
    point = data.get('point')

    # ✅ Validar subset
    if subset not in VALID_SUBSETS:
        return jsonify({"error": "Invalid subset"}), 400

    # ✅ Validar point
    if not point or 'x' not in point or 'y' not in point:
        return jsonify({"error": "Invalid point data"}), 400
    if not (0 <= point['x'] <= 1 and 0 <= point['y'] <= 1):
        return jsonify({"error": "Point coordinates must be normalized [0,1]"}), 400

    # ✅ Validar path seguro
    img_path = safe_resolve(DATASET_ROOT / subset / "images", filename)
    if img_path is None or not img_path.exists():
        return jsonify({"error": "Image not found"}), 404

    # ✅ Manejo de error si cv2 no puede leer la imagen
    img = cv2.imread(str(img_path))
    if img is None:
        return jsonify({"error": "Could not read image"}), 500

    h, w = img.shape[:2]
    input_point = [[point['x'] * w, point['y'] * h]]

    sam = get_sam_model()
    results = sam.predict(img, points=input_point, labels=[1], verbose=False)

    if results and len(results[0].masks.xyn) > 0:
        poly = results[0].masks.xyn[0].tolist()
        return jsonify({"polygon": poly})

    return jsonify({"error": "No mask found"}), 400


@app.route('/api/images/<subset>')
def list_images(subset):
    # ✅ Validar subset
    if subset not in VALID_SUBSETS:
        return jsonify({"error": "Invalid subset"}), 400

    img_dir = DATASET_ROOT / subset / "images"
    if not img_dir.exists():
        return jsonify({"error": "Images directory not found"}), 404

    images = [f.name for f in img_dir.glob("*.jpg")]
    return jsonify(sorted(images))


@app.route('/api/image/<subset>/<path:filename>')
def get_image(subset, filename):
    # ✅ Validar subset
    if subset not in VALID_SUBSETS:
        return jsonify({"error": "Invalid subset"}), 400

    safe_path = (DATASET_ROOT / subset / "images").absolute()
    return send_from_directory(str(safe_path), filename)


@app.route('/api/predict/<subset>/<path:filename>')
def predict(subset, filename):
    # ✅ Validar subset
    if subset not in VALID_SUBSETS:
        return jsonify({"error": "Invalid subset"}), 400

    # ✅ Path seguro
    img_path = safe_resolve(DATASET_ROOT / subset / "images", filename)
    if img_path is None or not img_path.exists():
        return "Image not found", 404

    results = model.predict(img_path, conf=0.15, verbose=False)[0]

    # ✅ Fichero temporal único por request (antes era tmp_pred.jpg compartido)
    pred_filename = f"pred_{uuid.uuid4().hex}.jpg"
    pred_path = Path(tempfile.gettempdir()) / pred_filename

    img_plotted = results.plot(boxes=True, masks=True)
    cv2.imwrite(str(pred_path), img_plotted)

    # ✅ Limpiar el fichero temporal después de enviarlo
    @after_this_request
    def cleanup(response):
        try:
            os.remove(pred_path)
        except Exception:
            pass
        return response

    return send_from_directory(str(pred_path.parent), pred_filename)


@app.route('/api/save_labels', methods=['POST'])
def save_labels():
    data = request.json
    subset = data.get('subset')
    filename = data.get('filename')
    labels = data.get('labels')

    # ✅ Validar subset
    if subset not in VALID_SUBSETS:
        return jsonify({"error": "Invalid subset"}), 400

    # ✅ Validar labels
    if labels is None or not isinstance(labels, list):
        return jsonify({"error": "Invalid labels format"}), 400

    # ✅ Path seguro contra path traversal
    label_filename = filename.replace(".jpg", ".txt")
    lab_path = safe_resolve(DATASET_ROOT / subset / "labels", label_filename)
    if lab_path is None:
        return jsonify({"error": "Invalid file path"}), 403

    # ✅ Crear directorio si no existe
    lab_path.parent.mkdir(parents=True, exist_ok=True)

    with open(lab_path, 'w') as f:
        f.write("\n".join(labels))

    return jsonify({"status": "success"})


if __name__ == '__main__':
    app.run(port=5000, debug=True)

from ultralytics import YOLO
from pathlib import Path
import os
import torch

# ✅ Paths via variable de entorno para portabilidad
BASE = Path(os.environ.get("APPED_ROOT", "C:/apped"))
DATA_YAML = BASE / "04_Datasets_Entrenamiento/super_focused_50/data.yaml"
OUTPUT_DIR = BASE / "train_yolo/runs/detect/SuperFocused_v1"


def train():
    # ✅ Validar que el yaml existe antes de empezar
    assert DATA_YAML.exists(), f"❌ data.yaml no encontrado: {DATA_YAML}"

    # ✅ Info de GPU antes de entrenar
    if torch.cuda.is_available():
        print(f"🖥️  CUDA disponible - GPU: {torch.cuda.get_device_name(0)}")
        print(f"💾 VRAM: {torch.cuda.get_device_properties(0).total_memory / 1e9:.1f} GB")
    else:
        print("⚠️  CUDA no disponible, entrenando en CPU (será lento)")

    print(f"📁 Dataset: {DATA_YAML}")
    print("🚀 Iniciando entrenamiento SUPER FOCUSED (50 imágenes)...")

    model = YOLO("yolo11m-seg.pt")

    results = model.train(
        data=str(DATA_YAML),
        epochs=100,
        imgsz=640,
        batch=-1,           # ✅ Auto-detect batch size según VRAM disponible (antes era 4 fijo)
        workers=0,
        project=str(OUTPUT_DIR.parent),
        name=OUTPUT_DIR.name,
        exist_ok=True,
        optimizer='AdamW',
        augment=True,
        patience=20,
        save=True,
        plots=True,
        verbose=True,
        # ✅ Se elimina task='segment' (redundante, el modelo -seg ya lo implica)
    )

    # ✅ Mostrar métricas finales
    print("\n✅ Entrenamiento completado.")
    map50 = results.results_dict.get('metrics/mAP50(B)', 'N/A')
    map50_95 = results.results_dict.get('metrics/mAP50-95(B)', 'N/A')
    print(f"📊 mAP50:    {map50}")
    print(f"📊 mAP50-95: {map50_95}")
    print(f"💾 Modelo guardado en: {OUTPUT_DIR / 'weights/best.pt'}")


if __name__ == "__main__":
    train()
